package lv.id.jc.piglatin.domain;

import java.util.regex.MatchResult;

import org.springframework.validation.annotation.Validated;

import jakarta.validation.constraints.NotBlank;
import jakarta.validation.constraints.Pattern;

@Validated
public record Word(@NotBlank @Pattern(regexp = "\\p{L}+") String value) {

    public static Word from(MatchResult result) {
        return new Word(result.group());
    }

    @Validated
    public static Word from(@NotBlank @Pattern(regexp = "\\p{L}+") String value) {
        return new Word(value);
    }
}
