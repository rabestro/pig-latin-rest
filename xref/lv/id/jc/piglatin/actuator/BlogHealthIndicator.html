3   import java.util.function.IntFunction;
4   import java.util.function.IntSupplier;
5   
6   import org.springframework.boot.actuate.health.Health;
7   import org.springframework.boot.actuate.health.HealthIndicator;
8   import org.springframework.stereotype.Component;
9   
10  /**
11   * The BlogHealthIndicator class is responsible for checking the health of a blog service by making a GET request to the blog's base URL.
12   * It implements the HealthIndicator interface provided by Spring Boot actuator library.
13   */
14  @Component
15  public class BlogHealthIndicator implements HealthIndicator {
16      private final IntFunction<Health> healthFunction;
17      private final IntSupplier statusCodeSupplier;
18  
19      public BlogHealthIndicator() {
20          this(new StatusCodeSupplier(), new HealthFunction());
21      }
22  
23      public BlogHealthIndicator(IntSupplier statusCodeSupplier, IntFunction<Health> healthFunction) {
24          this.healthFunction = healthFunction;
25          this.statusCodeSupplier = statusCodeSupplier;
26      }
27  
28      /**
29       * Returns the health status of the blog service.
30       * <p>
31       * The health status is determined by making a GET request to the blog's base URL
32       * and applying a health function to the HTTP response code. If the request is
33       * successful, the health function is applied to the response code and a Health
34       * object is returned. If an exception occurs during the request, a Health object
35       * is returned with a status of down and the exception as the detail.
36       *
37       * @return A Health object representing the health status of the blog service.
38       */
39      @Override
40      public Health health() {
41          try {
42              return healthFunction.apply(statusCodeSupplier.getAsInt());
43          } catch (RuntimeException e) {
44              return Health.down(e).withException(e.getCause()).build();
45          }
46      }
47  }
