3   import java.util.function.IntFunction;
4   
5   import org.springframework.boot.actuate.health.Health;
6   import org.springframework.http.HttpStatus;
7   import org.springframework.stereotype.Component;
8   
9   /**
10   * The HealthFunction class is an implementation of the IntFunction functional interface.
11   * It is responsible for converting an HTTP status code into a Health object that represents the health of a service.
12   */
13  @Component
14  public class HealthFunction implements IntFunction<Health> {
15      /**
16       * Converts an HTTP status code into a Health object that represents the health of a service.
17       *
18       * @param code the HTTP status code to be converted
19       * @return a Health object representing the health of the service
20       */
21      @Override
22      public Health apply(int code) {
23          if (code == HttpStatus.OK.value()) {
24              return Health.up().build();
25          }
26          return Health.down()
27              .withDetail("HTTP Status Code", code)
28              .build();
29      }
30  }
