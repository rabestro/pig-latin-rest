18  @Documented
19  @Retention(RUNTIME)
20  @Constraint(validatedBy = {})
21  @Target({PARAMETER, METHOD, FIELD})
22  @NotNull
23  @NotBlank
24  @Pattern(regexp = """
25      #
26      # A single word
27      # contains only Unicode letters and apostrophe (') in the middle
28      # only one apostrophe is allowed
29      #
30      ^               # Start of the line
31      (               # Start of the group
32          \\p{L}      # Any Unicode letter
33      |               # OR
34          (?<=\\p{L}) # Positive lookbehind for a Unicode letter
35          '           # An apostrophe
36          (?=\\p{L})  # Positive lookahead for a Unicode letter
37          (?!.*')     # Negative lookahead for another apostrophe
38      )               # End of the group
39      +               # One or more of the previous group
40      $               # End of the line
41      """,
42      flags = {Pattern.Flag.COMMENTS, Pattern.Flag.CASE_INSENSITIVE}
43  )
44  public @interface Word {
45      String message() default "must be a single word";
46  
47      Class<?>[] groups() default {};
48  
49      Class<? extends Payload>[] payload() default {};
50  }
