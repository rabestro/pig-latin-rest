3   import java.util.function.Function;
4   import java.util.regex.MatchResult;
5   import java.util.regex.Matcher;
6   
7   import org.springframework.beans.factory.annotation.Qualifier;
8   import org.springframework.stereotype.Component;
9   
10  @Component
11  public class PigLatinTranslator implements Translator {
12      private final Function<String, Matcher> wordMatcher;
13      private final Function<MatchResult, String> translator;
14  
15      public PigLatinTranslator(
16          Function<String, Matcher> wordMatcher,
17          @Qualifier("wordTranslator") Translator wordTranslator
18      ) {
19          this.wordMatcher = wordMatcher;
20          this.translator = ((Function<MatchResult, String>) MatchResult::group)
21              .andThen(wordTranslator);
22      }
23  
24      @Override
25      public String apply(String phrase) {
26          return wordMatcher.apply(phrase).replaceAll(translator);
27      }
28  }
