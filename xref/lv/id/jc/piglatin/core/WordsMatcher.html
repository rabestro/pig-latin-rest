3   import java.util.function.Function;
4   import java.util.regex.Matcher;
5   import java.util.regex.Pattern;
6   
7   import org.springframework.stereotype.Component;
8   
9   /**
10   * This class is responsible for matching words in a given phrase.
11   */
12  @Component
13  public class WordsMatcher implements Function<String, Matcher> {
14      private static final Pattern WORD = Pattern
15          .compile("""
16              # A word contains only Unicode letters and apostrophe (') in the middle
17              (               # Start of the group
18                  \\p{L}      # Any Unicode letter
19              |               # OR
20                  (?<=\\p{L}) # Positive lookbehind for a Unicode letter
21                  '           # An apostrophe
22                  (?=\\p{L})  # Positive lookahead for a Unicode letter
23                  (?!.*')     # Negative lookahead for another apostrophe
24              )+              # One or more of the previous group
25              """,
26              Pattern.COMMENTS + Pattern.CASE_INSENSITIVE);
27  
28      /**
29       * This method applies the word-matching pattern to a given phrase.
30       * It returns a Matcher object that can be used to find words in the phrase.
31       *
32       * @param phrase the phrase to apply the word-matching pattern to
33       * @return a Matcher object for finding words in the phrase
34       */
35      @Override
36      public Matcher apply(String phrase) {
37          return WORD.matcher(phrase);
38      }
39  }
